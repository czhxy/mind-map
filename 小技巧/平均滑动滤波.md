// 滑动滤波函数  5组位置 
#define WINDOWWID  5

typedef struct {
    int16_t slide_basedata[WINDOWWID];  // 滑动窗口缓冲区 
    int slide_pos;         // 当前位置 
} window_slide_param;  // 参数 


int16_t window_slide_input(window_slide_param* cur_slide,int16_t value)
{

  int16_t output = 0;

  if (cur_slide->slide_pos < WINDOWWID) //不满窗口，先填充
  {
    cur_slide->slide_basedata[cur_slide->slide_pos++] = value;
    output = value; //返回相同的值
  }
  else
  {
    int i = 0;
    int32_t sum = 0;

    memcpy(&cur_slide->slide_basedata[0], &cur_slide->slide_basedata[1], (WINDOWWID - 1) *sizeof(int16_t)); //将1之后的数据移到0之后，即移除头部
    cur_slide->slide_basedata[WINDOWWID - 1] = value;                       //即添加尾部
    
    for (i = 0; i < WINDOWWID; i++) //每一次都计算，可以避免累计浮点计算误差
      sum += cur_slide->slide_basedata[i];
    
    output = sum / WINDOWWID;
  }

  return output;
}
